% Preamble/Commands/LinearAlgebra/Multilinear.tex

\NewDocumentCommand{\TensorProduct}{o m m}
{
    \IfValueTF{#1}
    {
        #2 \otimes_{#1} #3
    }
    {
        #2 \otimes #3
    }
}

\NewDocumentCommand{\Hadamard}{m m}
{
    #1 \circ #2
}

\NewDocumentCommand{\KroneckerProduct}{o m m}
{
    \IfValueTF{#1}
    {
        #2 \otimes_{#1} #3
    }
    {
        #2 \otimes #3
    }
}

\NewDocumentCommand{\MultiLin}{o m m}
{
    \IfValueTF{#1}
    {
        \mathcal{L}_{#1} \left( #2; #3 \right)
    }
    {
        \mathcal{L} \left( #2; #3 \right)
    }
}

\NewDocumentCommand{\MultiForm}{o m m}
{
    \IfValueTF{#1}
    {
        \mathcal{L}^{#1} \left( #2 \right)
    }
    {
        \mathcal{L} \left( #2 \right)
    }
}

\NewDocumentCommand{\SymForm}{m m}
{
    \mathcal{S}^{#1} \left( #2 \right)
}

\NewDocumentCommand{\AltForm}{m m}
{
    \Lambda^{#1} \left( #2 \right)
}

\NewDocumentCommand{\Wedge}{m m}
{
    #1 \wedge #2
}
