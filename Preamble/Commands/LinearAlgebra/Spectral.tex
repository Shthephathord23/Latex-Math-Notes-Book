% Preamble/Commands/LinearAlgebra/Spectral.tex

\NewDocumentCommand{\Spectrum}{m}
{
    \sigma \left( #1 \right)
}

\NewDocumentCommand{\Eigenspace}{m m}
{
    E_{#2} \left( #1 \right)
}

\NewDocumentCommand{\GenEigenspace}{m m}
{
    G_{#2} \left( #1 \right)
}

\NewDocumentCommand{\MinPoly}{m}
{
    m_{#1}
}

\NewDocumentCommand{\CharPoly}{m}
{
    \chi_{#1}
}
